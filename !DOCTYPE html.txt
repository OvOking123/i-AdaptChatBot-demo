  <script>
    (() => {
      const widget = document.getElementById('ai-chat-widget');
      const messages = document.getElementById('chat-messages');
      const form = document.getElementById('chat-form');
      const input = document.getElementById('chat-input');
      const sendBtn = document.getElementById('chat-send-btn');

      // === REPLACE this with your actual n8n webhook URL ===
      const N8N_WEBHOOK_URL = 'https://YOUR_N8N_INSTANCE_URL/webhook/adaptive-ai-chatbot-http';

      // Customize your website context info here:
      const WEBSITE_CONTEXT = "Our demo company offers 24/7 support, 30-day returns, and a warranty on all products.";

      function appendMessage(text, sender = 'bot') {
        const msgDiv = document.createElement('div');
        msgDiv.classList.add('chat-message', sender);
        const innerDiv = document.createElement('div');
        innerDiv.textContent = text;
        msgDiv.appendChild(innerDiv);
        messages.appendChild(msgDiv);
        messages.scrollTop = messages.scrollHeight;
      }

      form.addEventListener('submit', async (e) => {
        e.preventDefault();
        const userText = input.value.trim();
        if (!userText) return;
        appendMessage(userText, 'user');
        input.value = '';
        sendBtn.disabled = true;

        try {
          const res = await fetch(N8N_WEBHOOK_URL, {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({
              message: userText,
              websiteData: WEBSITE_CONTEXT
            })
          });

          if (!res.ok) throw new Error('Network error');

          const json = await res.json();
          if (json.answer) {
            appendMessage(json.answer, 'bot');
          } else {
            appendMessage('Sorry, I did not get a response.', 'bot');
          }
        } catch (err) {
          appendMessage('Error: ' + err.message, 'bot');
        } finally {
          sendBtn.disabled = false;
        }
      });
    })();
  </script>
